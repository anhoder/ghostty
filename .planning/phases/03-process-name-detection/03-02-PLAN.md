```zig
fn termiosTimer(
    td_: ?*termio.Termio.ThreadData,
    _: *xev.Loop,
    _: *xev.Completion,
    r: xev.Timer.RunError!void,
) xev.CallbackAction {
    const td = td_.?;
    const exec = &td.backend.exec;

    // Existing termios mode detection
    const mode = (Pty{ .master = exec.read_thread_fd, .slave = undefined })
        .getMode() catch .{};

    if (!std.meta.eql(mode, exec.termios_mode)) {
        exec.termios_mode = mode;
        _ = td.surface_mailbox.push(.{
            .password_input = .{ .active = mode.echo == .off },
        }, .{ .forever = {} });
    }

    // Rearm timer
    if (exec.termios_timer_running) {
        exec.termios_timer.run(...);
    }
    return .disarm;
}
```

```zig
fn detectProcessName(
    td: *termio.Termio.ThreadData,
    pty_master_fd: posix.fd_t,
) !void {
    const name = try internal_os.process.getForegroundProcessName(
        td.arena.allocator(),
        pty_master_fd,
    ) orelse return;

    _ = td.surface_mailbox.push(.{
        .process_name_update = .{ .name = name },
    }, .{ .forever = {} });
}
```

3. Call `detectProcessName` in `termiosTimer` after password_input logic:

```zig
// NEW: Process name detection (non-Windows only)
if (comptime builtin.os.tag != .windows) {
    detectProcessName(td, exec.read_thread_fd) catch |err| {
        log.warn("process name detection failed err={}", .{err});
    };

```

**Why this approach:**
- Uses arena allocator from ThreadData (temporary allocation, cleaned up periodically)
- Sends every detection result (Surface will deduplicate if needed)
- Gracefully logs errors without crashing
- Windows excluded (no PTY support)

```zig
.process_name_update => |update| {
    defer self.alloc.free(update.name);

    // Free old process name if exists
    if (self.runtime_context.process_name) |old| {
        self.alloc.free(old);
    }

    // Store new process name (duplicate to Surface's allocator)
    self.runtime_context.process_name = try self.alloc.dupe(u8, update.name);
},
```

**Why this approach:**
- `defer` ensures message-owned string is freed
- Explicit free of old value prevents memory leak
- Duplicate to Surface's allocator (message uses arena allocator from I/O thread)
- Follows existing pattern from `pwd_change` handler
- No deduplication needed — updates are infrequent (~200ms polling, only changes sent)

**Note:** If `runtime_context.process_name` is not yet initialized as a field on Surface, it was added in Phase 2 (02-01). Verify the field exists: `runtime_context: input.Binding.RuntimeContext = .{}`
